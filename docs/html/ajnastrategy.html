Structural Patterns

Pattern: Registry-Based

function deployAjnaStrategy(
    address creatorToken,
    address vault
) external returns (address strategy) {
    strategy = new AjnaStrategy(vault, creatorToken...);
    address pool = _getOrDeployAjnaPool(creatorToken);
    AjnaStrategy(strategy).setAjnaPool(pool);
    IVault(vault).addStrategy(strategy, 100);
}
